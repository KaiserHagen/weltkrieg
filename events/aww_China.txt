# Germany declares war on Japan // China could join
country_event = {
	id = aw_chi.1
	title = aw_chi.1.t
	desc = aw_chi.1.d
	picture = GFX_report_event_gathering_protest
	
	fire_only_once = yes
	
	is_triggered_only = no
	
	mean_time_to_happen = {
		days=14
	}
	trigger = {
		tag = CHI 
		JAP = {
			OR = {
				has_completed_focus = remilitarization_of_korea_jap
				has_completed_focus = negotiations_with_germany_jap
			}
			has_defensive_war_with = GER
			
			NOT = { has_war_with = CHI }
		}
		has_global_flag = germany_attacks_japan
	}
	
	option = {
		name = aw_chi.1.a 	# Join War
		ai_chance = {
			factor = 40
			modifier = {
				factor = 0 
				CHI = { has_war = yes }
			}
		}
		JAP = {
			add_opinion_modifier = { 
				target = CHI
				modifier = chinese_support_against_germany
			}
		}
		CHI = {
			add_opinion_modifier = { 
				target = JAP
				modifier = chinese_support_against_germany
			}
		}
		declare_war_on = {
			target = GER
			type = take_state_focus
			generator = { 745 524 591 } 
		}
	}
	option = {
		name = aw_chi.1.b 	# don't join
		ai_chance = {
			factor = 60
			modifier = {
				factor = 100 
				CHI = { has_war = yes }
			}
		}
	}
}

##### Communist militia overthrow government
country_event = {
	id = aw_chi.38
	title = aw_chi.38.t
	desc = aw_chi.38.d
	picture = GFX_report_event_chinese_soldiers_fighting
	
	trigger = {
		tag = CHI
		CHI = {
			has_completed_focus = communist_militia_chi
			has_civil_war = no
			communism > 0.29
		}
	}

	mean_time_to_happen = { days = 5 }

	fire_only_once = yes

	#communists gain power
	option = {
		name = aw_chi.38.a
		ai_chance = {
			factor = 100
		}
		CHI = {
			set_cosmetic_tag = CHI_PRC
			set_politics = {
				ruling_party = communism
				elections_allowed = no
			}
			add_popularity = {
				ideology = communism
				popularity = 0.2
			}
		}
	}
}
