#Monarchists in Government
country_event = {
	id = election.18
	title = election.18.t
	desc = election.18.d
	picture = GFX_report_event_fascist_gathering
	
	is_triggered_only = yes

	trigger = {
		NOT = { tag = USA }
		OR = { NOT = { tag = SPR } NOT = { date < 1937.1.1 } }
		has_government = democratic
		monarchism > 0.15
	}
	
	option = {
		name = election.18.a
		add_popularity = {
			ideology = monarchism
			popularity = 0.1
		}
		set_country_flag = coalition_with_monarchists
	}

	option = {
		name = election.18.b
		add_stability = -0.05
		set_country_flag = government_declined_monarchists
	}
}

# Civil War for Monarchism
country_event = {
	id = political_w.4
	title = political_w.4.t
	desc = political_w.4.d
	picture = GFX_report_event_fascist_gathering
	
	trigger = {
		has_idea_with_trait = monarchist_demagogue
		NOT = { has_government = monarchism }
		monarchism > 0.25
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.75
			has_country_flag = monarchism_popular_movement
		}
	}
	
	option = {
		name = political_w.4.a
		if = {
			limit = {
				has_government = democratic
			}
			if = {
				limit = { monarchism > 0.4 }
				start_civil_war = {
					ruling_party = monarchism
					ideology = democratic
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { monarchism > 0.4 } }
				start_civil_war = {
					ruling_party = monarchism
					ideology = democratic
					size = 0.7
				}
			}
		}
		if = {
			limit = {
				has_government = communism
			}
			if = {
				limit = { monarchism > 0.4 }
				start_civil_war = {
					ruling_party = monarchism
					ideology = communism
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { monarchism > 0.4 } }
				start_civil_war = {
					ruling_party = monarchism
					ideology = communism
					size = 0.7
				}
			}
		}
		if = {
			limit = {
				has_government = neutrality
			}
			if = {
				limit = { monarchism > 0.4 }
				start_civil_war = {
					ruling_party = monarchism
					ideology = neutrality
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { monarchism > 0.4 } }
				start_civil_war = {
					ruling_party = monarchism
					ideology = neutrality
					size = 0.7
				}
			}
		}
		remove_ideas = monarchist_revolutionaries
		add_timed_idea = { idea = reign_of_terror days = 730 }
	}
	option = {
		name = political_w.4.b
	}
}
