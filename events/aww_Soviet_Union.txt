#France questions Treaty of Nuernberg
country_event = {
	id = aw_sov.16
	title = aw_sov.16.t
	desc = aw_sov.16.d
	picture = GFX_report_event_generic_sign_treaty1
	
	is_triggered_only = yes
	
	option = { # approve
		name = aw_sov.16.a
		ai_chance = {
			factor = 80
			modifier = {
				factor = 0
				SOV = {
					has_opinion = {
						target = FRA 
						value < -50
					}
				}
			}
		}
		add_opinion_modifier = {
			target = FRA
			modifier = question_treaty_of_nuernberg_approval
		}
		FRA = {
			add_opinion_modifier = {
				target = SOV
				modifier = question_treaty_of_nuernberg_approval
			}
		}
		if = {
			limit = {
				GER = {
					has_country_flag = qton_negative_reaction
					NOT = { has_opinion_modifier = question_treaty_of_nuernberg_reaction_sov }
				}
			}
			GER = {
				add_opinion_modifier = {
					target = SOV
					modifier = question_treaty_of_nuernberg_reaction_sov
				}
			}
		}
		set_country_flag = qton_approval_ita
	}
	
	option = { # ignore
		name = aw_sov.16.b
		ai_chance = {
			factor = 20
			modifier = {
				factor = 100
				SOV = {
					has_opinion = {
						target = FRA 
						value < -50
					}
				}
			}
		}
	}
	
}
