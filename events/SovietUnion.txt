# Successful Raid

	id = soviet.2
	title = soviet.2.t
	desc = soviet.2.d
	picture = GFX_report_event_dead_soldiers
	
	is_triggered_only = yes
	
	option = {
		name = soviet.2.a
		if = {
			limit = {
				561 = { 
					is_owned_by = SOV 
					any_neighbor_state = { is_owned_by = JAP } 
				}
			}
			561 = { add_manpower = -30000 }
		}		
		if = {
			limit = { 
				NOT = { 
					561 = { 
						is_owned_by = SOV 
						any_neighbor_state = { is_owned_by = JAP }
					}
				}
			}
			random_owned_state = {
				limit = { any_neighbor_state = { is_owned_by = JAP } }
				add_manpower = -30000
			}
		}

		army_experience = 10
	}
}
