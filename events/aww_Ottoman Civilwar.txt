country_event = {
	id = osman.6
	title = osman.6.t
	desc = osman.6.d
	picture = GFX_report_event_german_reichstag_gathering
	
	fire_only_once = yes
	
	trigger = {
		has_global_flag = ottoman_civil_war
		NOT = { has_global_flag = ottoman_civil_war_over }
		tag = OSM
		TUR = { 
			OR = {
				has_capitulated = yes
				exists = no
				is_puppet = yes
			}
		}
		OR = {
			IRQ = {
				OR = {
					surrender_progress > 0.33
					has_capitulated = yes
					is_puppet = yes
					exists = no
				}
			}
			AND = {
				SYR = {
					OR = {
						surrender_progress > 0.54
						has_capitulated = yes
						is_puppet = yes
						exists = no
					}
				}
				KUD = {
					OR = {
						surrender_progress > 0.54
						has_capitulated = yes
						is_puppet = yes
						exists = no
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 1	
	}
	
	immediate = {
		hidden_effect = {
			set_global_flag = ottoman_civil_war_over
			set_global_flag = ottoman_wins_civil_war
			OSM = {
				if = {
					limit = {
						SAU = {
							exists = yes
							is_in_faction_with = IRQ
							has_war_with = OSM
						}
					}
					white_peace = SAU
					puppet = SAU
				}
				white_peace = SYR
				white_peace = KUD
				white_peace = IRQ
				add_timed_idea = { idea = recover_from_civil_war_osm days = 365 }
			}
		}
	}
	
	option = { # unite the empire
		name = osman.6.a
		ai_chance = {
			factor = 60
			modifier = {
				factor = 0
				OR = {
					SYR = { is_ai = no }
					IRQ = { is_ai = no }
					KUD = { is_ai = no }
				}
			}
		}
		if = {
			limit = { TUR = { exists = yes }}
			annex_country = { target = TUR transfer_troops = yes }
		}
		if = {
			limit = {
				SAU = {
					is_puppet_of = OSM
				}
			}
			transfer_state = 772
			transfer_state = 744
			transfer_state = 674
			transfer_state = 658
		}
		annex_country = { target = SYR transfer_troops = yes }
		annex_country = { target = KUD transfer_troops = yes }
		annex_country = { target = IRQ transfer_troops = yes }
	}
	
	option = { # puppet
		name = osman.6.b
		ai_chance = {
			factor = 40
			modifier = {
				factor = 100
				OR = {
					SYR = { is_ai = no }
					IRQ = { is_ai = no }
					KUD = { is_ai = no }
				}
			}
		}
		transfer_state = 784
		transfer_state = 781
		transfer_state = 353
		if = {
			limit = { TUR = { exists = yes }}
			annex_country = { target = TUR transfer_troops = yes }
		}
		puppet = SYR
		puppet = KUD
		puppet = IRQ
		OSM = { add_stability = 0.10 }
