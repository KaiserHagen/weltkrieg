	id = usg.45
	title = usg.45.t
	desc = usg.45.d
	picture = GFX_report_event_generic_sign_treaty1
	
  fire_only_once = yes
  
  hidden = yes
	
	is_triggered_only = no
	
	mean_time_to_happen = {
		days=1
	}
  
	trigger = {
		tag = USG
    has_country_flag = grossdeutsche_loesung
    NOT = { country_exists = GER }
    USG = { has_war = no }
	}
	
	option = { # Accept
		name = usg.45.a
		ai_chance = { factor = 99 }
		every_state = {
      limit = {
        is_core_of = GER
        is_owned_by = USG
      }
      add_core_of = USG
    }
	 }
}
