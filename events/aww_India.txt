# Independence of Pakistan
country_event = {
	id = aw_india.7
	title = aw_india.7.t
	desc = aw_india.7.d
	picture = GFX_report_event_gathering_protest
	
	fire_only_once = yes
	
	trigger = {
		tag = RAJ
		date > 1937.1.7
		has_country_flag = pakistani_independence
		NOT = { country_exists = PAK }
	}

	mean_time_to_happen = {
		days = 2	
	}
	
	immediate = {
		hidden_effect = {
			PAK = {
				transfer_state = 440
				transfer_state = 445
				transfer_state = 442
				transfer_state = 443
				transfer_state = 444
				transfer_state = 430
				transfer_state = 441
				
				set_capital = 440
				set_stability = 0.65
				add_ideas = {
					limited_conscription
					civilian_economy
				}
				add_political_power = 50
				remove_ideas = freeing_pakistan_RAJ
			}
			set_country_flag = pakistan_is_free
		}
	}
	
	# free Pakistan
	option = {
		name = aw_india.7.a
		ai_chance = {
			factor = 100
		}
		give_guarantee = PAK
	}
	# play as Pakistan
	option = {
		name = aw_india.7.b
		ai_chance = {
			factor = 0
		}
		give_guarantee = PAK
		PAK = { change_tag_from = RAJ }
	}
}
